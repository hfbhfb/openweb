# 不完整的值赋值给接口

package main
import (
	"bytes"
	"io"
	"log"
	"reflect"
	"unsafe"
)
func main() {
	var w *bytes.Buffer
	f(w)
}
func f(r io.Reader) {
	log.Println(r)
	if r != nil {
		log.Println("not nil")
	}
	log.Println(unsafe.Sizeof(r))
	log.Println(reflect.TypeOf(r))
}

# 接口(go)的值拷贝: 代码路径

// 代码所在目录 /Users/hfb/projects/go/gopath/src_only_mod_copy/gopl.io/ch1/helloworld/mainvaluecopy.go
package main

import (
	"log"
	"reflect"
	"unsafe"
)

type I interface {
	tt()
	pp()
}

func (a *A) tt() {
	println(&(a.b))
}

//func (a A) tt() {
//	println(&(a.b))
//}

func (a A) pp() {
	println(&(a.b))
}

type A struct {
	b *B
}

type B struct {
	bb string
}

func testAny(a interface{}) {
	println(a)
	log.Println(unsafe.Sizeof(a))
	log.Println(reflect.TypeOf(a))

	switch v := a.(type) {
	case A:
		println(&(v.b))
		v.pp()
	}
}

func test(a I) {
	println(a)
	a.tt()
}
func test1(a *I) {
	aa := *a
	aa.tt()
	aa.pp()
}

func NewI() I {
	return &A{
		b: &B{bb: "bb"},
	}
}
func NewIRetValu() A {
	return A{
		b: &B{bb: "bb"},
	}
}

type C struct {
	i I
}

func main() {
	a := NewI()

	//switch v := a.(type) {
	//case *A:
	//	println(&(v.b))
	//	v.pp()
	//}
	c := &C{i: a}
	test(a)
	println("===========test1==========")
	test1(&a)
	c.i.tt()

	println("=====================")
	mValue := NewIRetValu()
	mValue.tt()
	var pI I = &mValue
	test(&mValue)
	test1(&pI)

	println("==========interface value===========")
	//test(mValue)   tt函数只有指针的选择器实现,报编译错误
	testAny(mValue)
}
